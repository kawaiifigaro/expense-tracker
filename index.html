function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    // 担当者シートを作成（初回のみ）
    var sheet = ss.getSheetByName(data.person);
    if (!sheet) {
      sheet = ss.insertSheet(data.person);
      var hdr = sheet.getRange(1, 1, 1, 8);
      hdr.setValues([['日付','金額（円）','支払先','内容','仕訳項目','備考','申請者','提出日時']]);
      hdr.setFontWeight('bold').setBackground('#1a73e8').setFontColor('#ffffff');
      sheet.setFrozenRows(1);
    }
    sheet.appendRow([
      data.date, Number(data.amount), data.payee,
      data.description, data.category, data.notes || '',
      data.person,
      Utilities.formatDate(new Date(), 'Asia/Tokyo', 'yyyy/MM/dd HH:mm:ss')
    ]);

    updateSummary(ss);

    return ContentService
      .createTextOutput(JSON.stringify({success: true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({success: false, error: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function updateSummary(ss) {
  var sheets = ss.getSheets();
  var summary = ss.getSheetByName('集計');
  if (!summary) summary = ss.insertSheet('集計', 0);
  summary.clearContents();

  var rows = [['担当者', '件数', '合計金額（円）', '最終更新']];
  var totalCount = 0, totalAmount = 0;

  sheets.forEach(function(s) {
    if (s.getName() === '集計') return;
    var vals = s.getDataRange().getValues().slice(1).filter(function(r){ return r[0]; });
    var count = vals.length;
    var amount = vals.reduce(function(a, r){ return a + (Number(r[1]) || 0); }, 0);
    totalCount += count;
    totalAmount += amount;
    rows.push([s.getName(), count, amount,
      Utilities.formatDate(new Date(), 'Asia/Tokyo', 'yyyy/MM/dd')]);
  });

  rows.push(['', '', '', '']);
  rows.push(['合計', totalCount, totalAmount, '']);

  summary.getRange(1, 1, rows.length, 4).setValues(rows);
  summary.getRange(1, 1, 1, 4).setFontWeight('bold').setBackground('#1a73e8').setFontColor('#ffffff');
  summary.getRange(rows.length, 1, 1, 3).setFontWeight('bold');
  summary.getRange(2, 3, rows.length - 1, 1).setNumberFormat('¥#,##0');
}
